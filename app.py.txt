import streamlit as st
import numpy as np
import pandas as pd
import yfinance as yf
import requests

st.title("Pattern Prediction Engine")

# --- YOUR PATTERN ---
num = "104281421"
digits = np.array([int(d) for d in num])
pattern = digits / max(digits)

# --- SELECT ASSET ---
ticker = st.text_input("Enter Ticker (ex: SPY, TSLA, BTC-USD)", "SPY")

# --- LOAD DATA ---
data = yf.download(ticker, period="5d", interval="5m")

if data.empty:
    st.error("No data found")
else:
    returns = data["Close"].pct_change().fillna(0)
    returns_norm = (returns - returns.min()) / (returns.max() - returns.min() + 1e-9)

    pattern_resized = np.interp(
        np.linspace(0, len(pattern), len(returns_norm)),
        np.arange(len(pattern)),
        pattern
    )

    alignment = np.correlate(returns_norm, pattern_resized, mode='same')
    alignment = (alignment - alignment.min()) / (alignment.max() - alignment.min() + 1e-9)

    st.line_chart(pd.DataFrame({
        "Market": returns_norm.values,
        "Pattern": pattern_resized,
        "Alignment": alignment
    }))

    # --- PREDICTION LOGIC ---
    threshold = 0.85
    future_window = 3

    results = []

    for i in range(len(alignment) - future_window):
        if alignment[i] > threshold:
            future_move = returns.iloc[i+1:i+1+future_window].mean()
            results.append(future_move)

    if results:
        wins = [r for r in results if r > 0]
        win_rate = len(wins) / len(results)
        avg_return = np.mean(results)

        st.write(f"Win Rate: {win_rate:.2f}")
        st.write(f"Avg Return: {avg_return:.5f}")
        st.write(f"Signals: {len(results)}")

    # --- ALERT ---
    latest_alignment = alignment[-1]

    if latest_alignment > threshold:
        st.warning("⚡ HIGH ALIGNMENT DETECTED")

        def send_alert(msg):
            url = "PASTE_DISCORD_WEBHOOK_HERE"
            requests.post(url, json={"content": msg})

        send_alert(f"{ticker} HIGH ALIGNMENT: {latest_alignment:.2f}")
